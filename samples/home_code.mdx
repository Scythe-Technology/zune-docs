```luau copy showLineNumbers filename="server.luau"
local net = require("@zcore/net")

local ok, serverHandle = net.serve({
     port = 8080,
     request = function(req)
          return "Hello!";
     end
})

-- Check if the server started successfully
assert(ok, serverHandle)

print("Http Server Running.")
```
```sh
$ zune run server
```
```ansi
Http Server Running.
```
